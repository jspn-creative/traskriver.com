```typescript
import { env } from '$env/dynamic/private';
import { getRequestEvent, query } from '$app/server';
import { hasActiveSubscription, createSubscriptionCookie } from '$lib/server/subscription';

export type StreamInfo = {
	liveHlsUrl: string;
	customerCode: string;
	inputId: string;
};

export const getStreamInfo = query(async () => {
	const event = getRequestEvent();
	const hasAccess = await hasActiveSubscription(event.cookies.get('subscription'));

	if (!hasAccess) {
		const cookieValue = await createSubscriptionCookie();
		event.cookies.set('subscription', cookieValue, { path: '/' });
	}

	const customer = env.CF_STREAM_CUSTOMER_CODE;
	const uid = env.CF_STREAM_LIVE_INPUT_UID;

	if (!customer?.trim() || !uid?.trim()) {
		throw new Error('Stream is not configured');
	}

	const liveHlsUrl = `https://customer-${customer}.cloudflarestream.com/${uid}/manifest/video.m3u8`;

	return { liveHlsUrl, customerCode: customer, inputId: uid } satisfies StreamInfo;
});
```

CF Stream Signed URL format (token replaces the live input UID):

```
https://customer-{customerCode}.cloudflarestream.com/{TOKEN}/manifest/video.m3u8
```

RS256 JWT structure:

- Header (base64url): `{"alg":"RS256","typ":"JWT","kid":"{keyId}"}`
- Payload (base64url): `{"sub":"{liveInputUid}","kid":"{keyId}","exp":{Math.floor(Date.now()/1000)+3600}}`
- Signature: RS256 sign of `{encodedHeader}.{encodedPayload}` using the imported JWK key
- Token: `{encodedHeader}.{encodedPayload}.{encodedSignature}`

For RS256 with Web Crypto API:

```typescript
// Import from JWK (CF_STREAM_SIGNING_JWK is a base64-encoded JWK string)
const jwkJson = JSON.parse(atob(jwkString)); // decode base64 → parse JSON
const privateKey = await crypto.subtle.importKey(
	'jwk',
	jwkJson,
	{ name: 'RSASSA-PKCS1-v1_5', hash: 'SHA-256' },
	false,
	['sign']
);

// Sign
const signature = await crypto.subtle.sign(
	'RSASSA-PKCS1-v1_5',
	privateKey,
	encoder.encode(`${encodedHeader}.${encodedPayload}`)
);

// base64url encode signature (ArrayBuffer → Uint8Array → string → base64url)
```

   ```typescript
   	uid: string,

   	jwkBase64: string
   ): Promise<string> => {
   	const jwkJson = JSON.parse(atob(jwkBase64));

   		'jwk',
   		jwkJson,
   		{ name: 'RSASSA-PKCS1-v1_5', hash: 'SHA-256' },
   		false,
   		['sign']
   	);

   	const header = toBase64Url(JSON.stringify({ alg: 'RS256', typ: 'JWT', kid: keyId }));
   	const payload = toBase64Url(
   		JSON.stringify({ sub: uid, kid: keyId, exp: Math.floor(Date.now() / 1000) + 3600 })

   	const signature = await crypto.subtle.sign(
   		'RSASSA-PKCS1-v1_5',
   		privateKey,
   		encoder.encode(signingInput)
   	);
   	const sigString = String.fromCharCode(...new Uint8Array(signature));

   	return `${signingInput}.${toBase64Url(sigString)}`;
   };
   ```
