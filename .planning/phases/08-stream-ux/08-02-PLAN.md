    **Relay polling (D-02, D-03):**
    Create a polling mechanism using `$effect` and `setTimeout` chaining (not `setInterval`):

    ```typescript
    const POLL_INTERVAL_MS = 3_000;
    const STARTING_TIMEOUT_MS = 60_000;

    let polling = $state(false);
    let startingTimestamp = $state<number | null>(null);
    let startingUnavailableAccumulatedMs = $state(0);
    let startingUnavailableSince = $state<number | null>(null);
    let relayStale = $state(false);
    let lastKnownRelayState = $state<RelayStatusResponse['state']>(null);

    const pollRelayStatus = async () => {
      try {
        const res = await fetch('/api/relay/status');
        if (!res.ok) return;
        const data: RelayStatusResponse = await res.json();

        relayStale = data.stale;
        if (data.state !== null) {
          lastKnownRelayState = data.state;
        }

        // D-06: Starting timeout check (explicit around unavailable periods)
        // Policy: timeout counts only while relay is responsive; unavailable time is paused.
        if (phase === 'starting' && startingTimestamp) {
          const elapsed = Date.now() - startingTimestamp - startingUnavailableAccumulatedMs;
          if (elapsed > STARTING_TIMEOUT_MS) {
            phase = 'error';
            polling = false;
            return;
          }
        }

        if (data.stale) {
          // Review concern #1: if relay was last known stopped, prefer ended over unavailable.
          if (lastKnownRelayState === 'stopped' && (phase === 'starting' || phase === 'ended_confirming')) {
            phase = 'ended';
            polling = false;
            return;
          }

          // D-11: Relay offline — distinct from starting
          if (phase === 'starting') {
            phase = 'unavailable';
            if (!startingUnavailableSince) startingUnavailableSince = Date.now();
          }
          return;
        }

        // relay responsive again; close any paused unavailable window
        if (startingUnavailableSince) {
          startingUnavailableAccumulatedMs += Date.now() - startingUnavailableSince;
          startingUnavailableSince = null;
        }

        // Relay is responsive — drive transitions from relay state
        if (phase === 'starting' || phase === 'unavailable') {
          if (data.state === 'live') {
            phase = 'live'; // VideoPlayer will load and eventually fire onPlaying
          } else if (data.state === 'idle' || data.state === 'starting') {
            // Relay is responsive but not yet live — return to starting if we were unavailable
            if (phase === 'unavailable') {
              phase = 'starting';
            }
            // Otherwise stay in starting
          } else if (data.state === 'stopped') {
            // naming clarity: relay state "stopped" maps to UI phase "ended"
            phase = 'ended';
            polling = false;
          }
        } else if (phase === 'ended_confirming') {
          // D-07: Confirming stream end after HLS error
          if (data.state === 'idle' || data.state === 'stopped') {
            phase = 'ended';
            polling = false;
          } else if (data.state === 'live') {
            // False alarm — relay still live, maybe transient HLS hiccup
            phase = 'live';
            // polling continues until onPlaying stops it again
          }
        }
      } catch {
        // Network error during polling — non-fatal, will retry next tick
        log('relay status poll failed');
      }
    };
    ```

    Note: `ended_confirming` is an internal transient state used between HLS error and relay confirmation. Add it to the phase type:

| T-08-05 | Information Disclosure | Relay state in page UI | accept | State labels (starting/live/ended/unavailable) reveal operational status — this is intentional UX, not sensitive data. |
