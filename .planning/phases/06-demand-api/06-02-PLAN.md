    ```typescript
    export const POST = async ({ platform }) => {
      const kv = platform?.env?.RIVER_KV;
      if (!kv) throw error(503, 'Service unavailable');

      const now = Date.now();

      // Read-before-write throttling: skip write if last demand was <30s ago
      const existing = await kv.get(DEMAND_KEY);
      if (existing) {
        const lastTimestamp = parseInt(existing, 10);
        if (!isNaN(lastTimestamp) && now - lastTimestamp < THROTTLE_MS) {
          return json({ ok: true });
        }
      }

      // Write demand timestamp — NO expirationTtl per CONTEXT.md
      // "KV demand key has NO expirationTtl — key persists until overwritten by next demand click"
      await kv.put(DEMAND_KEY, now.toString());

      return json({ ok: true });
    };
    ```

    Key decisions:

    - Response is `{ ok: true }` per CONTEXT.md: "Demand endpoint returns minimal response: `{ ok: true }`"
    - NO `expirationTtl` on the KV key per CONTEXT.md: "KV demand key has NO expirationTtl — key persists until overwritten"
    - 30s read-before-write throttling per STATE.md key decision

    **GET handler (authenticated — relay polling):**
    ```typescript
    export const GET = async ({ request, platform }) => {
      const kv = platform?.env?.RIVER_KV;
      const token = platform?.env?.RELAY_API_TOKEN;
      if (!kv || !token) throw error(503, 'Service unavailable');

      // Bearer token auth — return 401 for missing/invalid token
      // Per CONTEXT.md: "401 Unauthorized for missing/invalid bearer token — don't leak endpoint purpose"
      const authHeader = request.headers.get('authorization');
      if (!authHeader || authHeader !== `Bearer ${token}`) {
        throw error(401, 'Unauthorized');
      }

      const demandStr = await kv.get(DEMAND_KEY);
      const now = Date.now();

      // Demand window is configurable, default 5 minutes (300,000ms)
      // Read DEMAND_WINDOW_MS from env if available, else default
      const windowMs = 5 * 60 * 1000; // 300,000ms = 5 minutes

      if (!demandStr) {
        return json({
          shouldStream: false,
          demandTimestamp: null,
          ttlSeconds: 0
        } satisfies DemandResponse);
      }

      const demandTimestamp = parseInt(demandStr, 10);
      if (isNaN(demandTimestamp)) {
        return json({
          shouldStream: false,
          demandTimestamp: null,
          ttlSeconds: 0
        } satisfies DemandResponse);
      }

      const elapsed = now - demandTimestamp;
      const shouldStream = elapsed < windowMs;
      const ttlSeconds = shouldStream ? Math.ceil((windowMs - elapsed) / 1000) : 0;

      return json({
        shouldStream,
        demandTimestamp,
        ttlSeconds
      } satisfies DemandResponse);
    };
    ```

    Key decisions:
    - GET requires bearer token per CONTEXT.md: "authenticated via bearer token (shared secret)"
    - Uses `satisfies DemandResponse` to ensure type-safe responses
    - `shouldStream` calculated from timestamp age vs window (not KV TTL) per CONTEXT.md: "Worker calculates expiry from stored KV timestamp (not KV TTL-based expiry)"
    - `ttlSeconds` included (Claude's discretion — useful for relay logging)
    - 401 response for auth failure per CONTEXT.md: "don't leak endpoint purpose"

    **POST handler (authenticated — relay state reporting):**
    ```typescript
    export const POST = async ({ request, platform }) => {
      const kv = platform?.env?.RIVER_KV;
      const token = platform?.env?.RELAY_API_TOKEN;
      if (!kv || !token) throw error(503, 'Service unavailable');

      // Bearer token auth — same pattern as demand GET
      const authHeader = request.headers.get('authorization');
      if (!authHeader || authHeader !== `Bearer ${token}`) {
        throw error(401, 'Unauthorized');
      }

      // Parse relay status payload
      let payload: RelayStatusPayload;
      try {
        payload = await request.json();
      } catch {
        throw error(400, 'Invalid JSON');
      }

      // Validate required fields (state and timestamp)
      if (typeof payload.state !== 'string' || !payload.state) {
        throw error(400, 'Missing state');
      }
      if (typeof payload.timestamp !== 'number' || !Number.isFinite(payload.timestamp)) {
        throw error(400, 'Missing timestamp');
      }

      // Store in KV with expirationTtl — stale timestamp detection
      // Per CONTEXT.md: "stale timestamp threshold — if relay-status timestamp is >2 min old, consider relay offline"
      // Using KV TTL as secondary expiry so stale entries auto-clean
      await kv.put(RELAY_STATUS_KEY, JSON.stringify(payload), {
        expirationTtl: STATUS_TTL_SECONDS
      });

      return json({ ok: true });
    };
    ```

    Key decisions:
    - Bearer token auth using same pattern as demand GET (per CONTEXT.md: "same shared secret as demand polling")
    - Accepts **any** string for `state` field (per CONTEXT.md: "accepts any state value (no enum validation) — flexible, relay may evolve")
    - Validates `state` is a non-empty string and `timestamp` is a finite number — basic input validation without enum restriction
    - KV TTL set to 120s (2 min) — per CONTEXT.md: "if relay-status timestamp is >2 min old, consider relay offline". KV TTL acts as automatic cleanup; the timestamp inside the value is the primary indicator
    - Response is `{ ok: true }` for consistency with demand POST
