    export class FfmpegManager {
      constructor(config: { rtspUrl: string; streamUrl: string });
      async start(): Promise<boolean>;
      async stop(): Promise<void>;
      isRunning(): boolean;
      onExit(callback: (code: number | null, signal: string | null) => void): void;

    **`start()` implementation:**
    - Spawn ffmpeg via `Bun.spawn(["ffmpeg", "-rtsp_transport", "tcp", "-i", config.rtspUrl, "-c:v", "copy", "-c:a", "aac", "-f", "flv", config.streamUrl], { stdout: "pipe", stderr: "pipe" })`
    - The ffmpeg args: `-rtsp_transport tcp` for reliable RTSP, `-i {rtspUrl}` input, `-c:v copy` passthrough video, `-c:a aac` transcode audio, `-f flv` output format for RTMPS, `{streamUrl}` output destination
    - Store reference in `this.process`
    - Set up exit handler: when process exits, call all registered `onExit` callbacks with `(exitCode, signalCode)`
    - To detect process exit with Bun.spawn: use `this.process.exited.then((code) => { ... })` — `Bun.spawn` returns a `Subprocess` with an `exited` promise
    - Log: `log.info("ffmpeg started (pid: {pid})")`
    - Return `true` if spawn succeeds, `false` if it throws
    - Guard against double-start: if `this.process` is already set and running, log warning and return false

    **`stop()` implementation:**
    - If no process running, return immediately
    - Send SIGTERM: `this.process.kill("SIGTERM")`
    - Log: `log.info("ffmpeg SIGTERM sent (pid: {pid})")`
    - Set a 10-second SIGKILL fallback timer: `setTimeout(() => { if (this.isRunning()) { this.process.kill("SIGKILL"); log.warn("ffmpeg SIGKILL sent (pid: {pid}) — SIGTERM timeout"); } }, 10_000)`
    - Await process exit: `await this.process.exited`
    - Clear the SIGKILL timer if process exited before timeout
    - Set `this.process = null` after exit
    - Log: `log.info("ffmpeg stopped (exit: {code})")`

    **`isRunning()` implementation:**
    - Check if `this.process` is not null and hasn't exited yet
    - Bun.spawn subprocess: check `this.process.exitCode === null` (null means still running)

    **`onExit()` implementation:**
    - Store callbacks in array, fire all when process exits
    - Callbacks receive `(exitCode, signalCode)` — null exitCode means killed by signal

    Import `log` from `./logger`.
    Do NOT import any Node.js `child_process` module — use Bun.spawn exclusively.

    **Main loop** using `setTimeout` chaining (NOT setInterval):
    ```typescript
    async function tick(): Promise<void> {
      const state = sm.getState();

      // --- Cooldown wait ---
      if (state === "cooldown") {
        // Wait for cooldown period, then transition to idle
        log.info(`cooldown: waiting ${COOLDOWN_MS / 1000}s before retry`);
        await new Promise((r) => setTimeout(r, COOLDOWN_MS));
        sm.transition("idle", "cooldown elapsed");
        scheduleTick();
        return;
      }

      // --- Poll for demand ---
      const result = await poller.poll();

      // --- Safety policy: too many consecutive failures ---
      if (result.consecutiveFailures >= config.failureThreshold) {
        if (state === "live" || state === "starting") {
          log.warn(`safety stop: ${result.consecutiveFailures} consecutive poll failures`);
          sm.transition("stopping", `${result.consecutiveFailures} consecutive poll failures`);
          await ffmpeg.stop();
          sm.transition("idle", "safety stop complete");
          poller.resetFailures();
        }
        scheduleTick();
        return;
      }

      // --- State-specific logic ---
      if (state === "idle" && result.shouldStream) {
        // Demand detected — start ffmpeg
        sm.transition("starting", "demand detected");
        const started = await ffmpeg.start();
        if (started) {
          sm.transition("live", "ffmpeg confirmed running");
        } else {
          sm.transition("cooldown", "ffmpeg failed to start");
        }
      } else if ((state === "live" || state === "starting") && !result.shouldStream) {
        // Demand expired — stop ffmpeg
        sm.transition("stopping", "demand expired");
        await ffmpeg.stop();
        sm.transition("idle", "clean stop complete");
      }
      // If idle && !shouldStream: stay idle, do nothing
      // If live && shouldStream: stay live, keep streaming

      scheduleTick();
    }

    function scheduleTick(): void {
      setTimeout(tick, config.pollIntervalMs);
    }
    ```

    **Entry point:**
    ```typescript
    async function run(): Promise<void> {
      log.info("relay starting");
      log.info(`config: poll=${config.pollIntervalMs}ms, timeout=${config.requestTimeoutMs}ms, failureThreshold=${config.failureThreshold}`);

      // Report initial idle state
      await reporter.report("idle");

      // Start the polling loop
      tick();
    }

    // Graceful shutdown on SIGTERM/SIGINT
    process.on("SIGTERM", async () => {
      log.info("SIGTERM received, shutting down");
      if (ffmpeg.isRunning()) {
        await ffmpeg.stop();
      }
      await reporter.report("stopped");
      process.exit(0);
    });

    process.on("SIGINT", async () => {
      log.info("SIGINT received, shutting down");
      if (ffmpeg.isRunning()) {
        await ffmpeg.stop();
      }
      await reporter.report("stopped");
      process.exit(0);
    });

    run().catch((err) => {
      log.error(`fatal: ${err}`);
      process.exit(1);
    });
    ```
